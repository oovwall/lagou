# 自动化构建

## Gulp的组合任务
gulpfile.js
```js
const { series, parallel } = require('gulp')

const task1 = done => {
  setTimeout(() => {
    console.log('task1 working')
    done()
  }, 1000)
}

const task2 = done => {
  setTimeout(() => {
    console.log('task2 working')
    done()
  }, 1000)
}

const task3 = done => {
  setTimeout(() => {
    console.log('task3 working')
    done()
  }, 1000)
}

// 串行任务，同步执行队列
exports.series = series(task1, task2, task3) 
// 并行任务，异步执行队列
exports.parallel = parallel(task1, task2, task3)
```
